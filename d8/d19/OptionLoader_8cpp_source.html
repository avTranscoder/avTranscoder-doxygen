222 OptionLoader::OptionArray OptionLoader::loadOptions( void* av_class, int req_flags )
223 {
224  OptionArray options;
225 
226  std::multimap<std::string, int> optionUnitToIndex;
227  std::vector<Option> childOptions;
228 
229  const AVOption* avOption = NULL;
230 
231  // iterate on options
232 #if LIBAVCODEC_VERSION_INT < AV_VERSION_INT( 51, 12, 0 )
233  while( ( avOption = av_next_option( av_class, avOption ) ) )
234 #else
235  while( ( avOption = av_opt_next( av_class, avOption ) ) )
236 #endif
237  {
238  if( !avOption ||
239  ! avOption->name ||
240  ( avOption->flags & req_flags ) != req_flags )
241  {
242  continue;
243  }
244 
245  OptionType optionType = Option::getTypeFromAVOption( avOption->unit, avOption->type );
246 
247  if( optionType == TypeChild )
248  {
249  childOptions.push_back( Option( *avOption, optionType ) );
250  }
251  else
252  {
253  options.push_back( Option( *avOption, optionType ) );
254  optionUnitToIndex.insert(
255  std::pair<std::string, int>(
256  std::string( avOption->unit ? avOption->unit : "" ),
257  options.size() - 1 )
258  );
259  }
260  }
261 
262  // iterate on child options
263  for( std::vector<Option>::iterator itOption = childOptions.begin(); itOption != childOptions.end(); ++itOption )
264  {
265  for( std::multimap<std::string, int>::iterator itUnit = optionUnitToIndex.begin(); itUnit != optionUnitToIndex.end(); ++itUnit )
266  {
267  if( itUnit->first == itOption->getUnit() )
268  {
269  int indexParentOption = itUnit->second;
270  Option& parentOption = options.at( indexParentOption );
271 
272  parentOption.appendChild( *itOption );
273 
274  // child of a Choice
275  if( parentOption.getType() == TypeChoice )
276  {
277  if( itOption->getDefaultValueInt() == parentOption.getDefaultValueInt() )
278  parentOption.setDefaultChildIndex( parentOption.getNbChilds() - 1 );
279  }
280  }
281  }
282  }
283  return options;
